# JARVIS backend dependencies — install into a venv at ./venv:
#   python -m venv venv
#   venv\Scripts\pip install -r requirements.txt   (Windows)
#   venv/bin/pip install -r requirements.txt        (macOS/Linux)

# Web server
fastapi
uvicorn[standard]
websockets

# Brain / vision / STT — all via Groq's free OpenAI-compatible API
openai

# Text-to-speech (Microsoft Edge neural voices, free, cloud)
edge-tts

# Voice input capture + VAD
sounddevice
numpy

# System tools
psutil

# Screen capture (for cloud vision)
mss
Pillow

# Market data for the ICT scanner (Indian markets: Nifty, Sensex, NSE stocks)
yfinance
pandas

# Video understanding: frame sampling (opencv) + URL download (yt-dlp);
# audio is transcribed by Groq Whisper, so no ffmpeg binary needed.
opencv-python-headless
yt-dlp

# Document uploads — PDF text extraction (docx/text use stdlib only)
pypdf

# Cross-model memory hub — MCP server so Claude/ChatGPT/Gemini share JARVIS's store
mcp

# Cortex vector index — semantic recall over facts + episodes. Falls back to a
# pure-Python in-RAM cosine index if this is missing, so JARVIS still boots.
chromadb

# Local compute layer (governor/device/consolidation/models_advisor) talks to Ollama
ollama

# Web search — optional faster DDG client (falls back to HTML scrape)
duckduckgo-search

# Optional fallback brain (only if you set ANTHROPIC_API_KEY)
anthropic
